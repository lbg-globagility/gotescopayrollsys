UPDATE paywithholdingtax t
SET t.EffectiveDateTo='2024-12-31',
t.LastUpd=CURRENT_TIMESTAMP(),
t.LastUpdBy=IFNULL(t.LastUpdBy, t.CreatedBy)
WHERE t.EffectiveDateFrom='2023-01-01' AND t.EffectiveDateTo='2023-12-31'
;

UPDATE paysocialsecurity ss

ss.LastUpd=CURRENT_TIMESTAMP()
WHERE ss.EffectiveDateFrom='2023-01-01' AND ss.EffectiveDateTo='2023-12-31'
;
